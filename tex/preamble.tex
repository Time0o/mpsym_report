\newcommand{\definetcolorbox}[2]{
  \newtcolorbox[
    auto counter,
    number within=chapter,
    number freestyle={\noexpand\thechapter.\noexpand\arabic{\tcbcounter}}]{#1}[2][]
  {
    enhanced,
    breakable,
    enlarge top by=1ex,
    enlarge bottom by=1ex,
    before upper={\parindent15pt\noindent},
    after={\noindent},
    fonttitle=\bfseries,
    title=#2~\thetcbcounter: ##2,
    ##1
  }
}
